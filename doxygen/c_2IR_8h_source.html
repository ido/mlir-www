30 /** Opaque type declarations.
31  *
32  * Types are exposed to C bindings as structs containing opaque pointers. They
33  * are not supposed to be inspected from C. This allows the underlying
34  * representation to change without affecting the API users. The use of structs
35  * instead of typedefs enables some type safety as structs are not implicitly
36  * convertible to each other.
37  *
38  * Instances of these types may or may not own the underlying object (most often
39  * only point to an IR fragment without owning it). The ownership semantics is
40  * defined by how an instance of the type was obtained.
41  */
42 /*============================================================================*/
43 
44 #define DEFINE_C_API_STRUCT(name, storage) \
45  struct name { \
46  storage *ptr; \
47  }; \
48  typedef struct name name

186 /** An auxiliary class for constructing operations.
187  *
188  * This class contains all the information necessary to construct the operation.
189  * It owns the MlirRegions it has pointers to and does not own anything else.
190  * By default, the state can be constructed from a name and location, the latter
191  * being also used to access the context, and has no other components. These
192  * components can be added progressively until the operation is constructed.
193  * Users are not expected to rely on the internals of this class and should use
194  * mlirOperationState* functions instead.
195  */
196 struct MlirOperationState {
197  const char *name;
198  MlirLocation location;
199  intptr_t nResults;
200  MlirType *results;
201  intptr_t nOperands;
202  MlirValue *operands;
203  intptr_t nRegions;
204  MlirRegion *regions;
205  intptr_t nSuccessors;
206  MlirBlock *successors;
207  intptr_t nAttributes;
208  MlirNamedAttribute *attributes;
209 };
210 typedef struct MlirOperationState MlirOperationState;
