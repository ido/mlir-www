27 /// Template for a reference to a concrete type which captures a python
28 /// reference to its underlying python object.
29 template <typename T>
30 class PyObjectRef {
31 public:
32  PyObjectRef(T *referrent, pybind11::object object)
33  : referrent(referrent), object(std::move(object)) {
34  assert(this->referrent &&
35  "cannot construct PyObjectRef with null referrent");
36  assert(this->object && "cannot construct PyObjectRef with null object");
37  }
38  PyObjectRef(PyObjectRef &&other)
39  : referrent(other.referrent), object(std::move(other.object)) {
40  other.referrent = nullptr;
41  assert(!other.object);
42  }
43  PyObjectRef(const PyObjectRef &other)
44  : referrent(other.referrent), object(other.object /* copies */) {}
45  ~PyObjectRef() {}
46 
47  int getRefCount() {
48  if (!object)
49  return 0;
50  return object.ref_count();
51  }
52 
53  /// Releases the object held by this instance, returning it.
54  /// This is the proper thing to return from a function that wants to return
55  /// the reference. Note that this does not work from initializers.
56  pybind11::object releaseObject() {
57  assert(referrent && object);
58  referrent = nullptr;
59  auto stolen = std::move(object);
60  return stolen;
61  }
62 
63  T *operator->() {
64  assert(referrent && object);
65  return referrent;
66  }
67  pybind11::object getObject() {
68  assert(referrent && object);
69  return object;
70  }
71  operator bool() const { return referrent && object; }
72 
73 private:
74  T *referrent;
75  pybind11::object object;
76 };

87  /// For the case of a python __init__ (py::init) method, pybind11 is quite
88  /// strict about needing to return a pointer that is not yet associated to
89  /// an py::object. Since the forContext() method acts like a pool, possibly
90  /// returning a recycled context, it does not satisfy this need. The usual
91  /// way in python to accomplish such a thing is to override __new__, but
92  /// that is also not supported by pybind11. Instead, we use this entry
93  /// point which always constructs a fresh context (which cannot alias an
94  /// existing one because it is fresh).
95  static PyMlirContext *createNewContextForInit();
