31 /** Attribute on MLIR Python objects that expose their C-API pointer.
32  * This will be a type-specific capsule created as per one of the helpers
33  * below.
34  *
35  * Ownership is not transferred by acquiring a capsule in this way: the
36  * validity of the pointer wrapped by the capsule will be bounded by the
37  * lifetime of the Python object that produced it. Only the name and pointer
38  * of the capsule are set. The caller is free to set a destructor and context
39  * as needed to manage anything further. */
40 #define MLIR_PYTHON_CAPI_PTR_ATTR "_CAPIPtr"
41 
42 /** Attribute on MLIR Python objects that exposes a factory function for
43  * constructing the corresponding Python object from a type-specific
44  * capsule wrapping the C-API pointer. The signature of the function is:
45  * def _CAPICreate(capsule) -> object
46  * Calling such a function implies a transfer of ownership of the object the
47  * capsule wraps: after such a call, the capsule should be considered invalid,
48  * and its wrapped pointer must not be destroyed.
49  *
50  * Only a very small number of Python objects can be created in such a fashion
51  * (i.e. top-level types such as Context where the lifetime can be cleanly
52  * delineated). */
53 #define MLIR_PYTHON_CAPI_FACTORY_ATTR "_CAPICreate"

59 /** Creates a capsule object encapsulating the raw C-API MlirContext.
60  * The returned capsule does not extend or affect ownership of any Python
61  * objects that reference the context in any way.
62  */
63 inline PyObject *mlirPythonContextToCapsule(MlirContext context) {
64  return PyCapsule_New(context.ptr, MLIR_PYTHON_CAPSULE_CONTEXT, NULL);
65 }
66 
67 /** Extracts a MlirContext from a capsule as produced from
68  * mlirPythonContextToCapsule. If the capsule is not of the right type, then
69  * a null context is returned (as checked via mlirContextIsNull). In such a
70  * case, the Python APIs will have already set an error. */
71 inline MlirContext mlirPythonCapsuleToContext(PyObject *capsule) {
72  void *ptr = PyCapsule_GetPointer(capsule, MLIR_PYTHON_CAPSULE_CONTEXT);
73  MlirContext context = {ptr};
74  return context;
75 }

77 /** Creates a capsule object encapsulating the raw C-API MlirModule.
78  * The returned capsule does not extend or affect ownership of any Python
79  * objects that reference the module in any way. */
80 inline PyObject *mlirPythonModuleToCapsule(MlirModule module) {
81 #ifdef __cplusplus
82  void *ptr = const_cast<void *>(module.ptr);
83 #else
84  void *ptr = (void *)ptr;
85 #endif
86  return PyCapsule_New(ptr, MLIR_PYTHON_CAPSULE_MODULE, NULL);
87 }
