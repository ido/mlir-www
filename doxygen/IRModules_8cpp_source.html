395  parentOperation->checkValid();
396  newOperation.checkValid();
397  if (index < 0) {
398  throw SetPyError(

403  throw SetPyError(

500 py::object PyMlirContext::createOperation(
501  std::string name, PyLocation location,
502  llvm::Optional<std::vector<PyType *>> results,
503  llvm::Optional<py::dict> attributes,
504  llvm::Optional<std::vector<PyBlock *>> successors, int regions) {
505  llvm::SmallVector<MlirType, 4> mlirResults;
506  llvm::SmallVector<MlirBlock, 4> mlirSuccessors;
507  llvm::SmallVector<std::pair<std::string, MlirAttribute>, 4> mlirAttributes;
508 
509  // General parameter validation.
510  if (regions < 0)
511  throw SetPyError(PyExc_ValueError, "number of regions must be >= 0");
512 
513  // Unpack/validate results.
514  if (results) {
515  mlirResults.reserve(results->size());
516  for (PyType *result : *results) {
517  // TODO: Verify result type originate from the same context.
518  if (!result)
519  throw SetPyError(PyExc_ValueError, "result type cannot be None");
520  mlirResults.push_back(result->type);
521  }
522  }
523  // Unpack/validate attributes.
524  if (attributes) {
525  mlirAttributes.reserve(attributes->size());
526  for (auto &it : *attributes) {
527 
528  auto name = it.first.cast<std::string>();
529  auto &attribute = it.second.cast<PyAttribute &>();
530  // TODO: Verify attribute originates from the same context.
531  mlirAttributes.emplace_back(std::move(name), attribute.attr);
532  }
533  }
534  // Unpack/validate successors.
535  if (successors) {
536  llvm::SmallVector<MlirBlock, 4> mlirSuccessors;
537  mlirSuccessors.reserve(successors->size());
538  for (auto *successor : *successors) {
539  // TODO: Verify successor originate from the same context.
540  if (!successor)
541  throw SetPyError(PyExc_ValueError, "successor block cannot be None");
542  mlirSuccessors.push_back(successor->get());
543  }
544  }
545 
546  // Apply unpacked/validated to the operation state. Beyond this
547  // point, exceptions cannot be thrown or else the state will leak.
548  MlirOperationState state = mlirOperationStateGet(name.c_str(), location.loc);
549  if (!mlirResults.empty())
550  mlirOperationStateAddResults(&state, mlirResults.size(),
551  mlirResults.data());
552  if (!mlirAttributes.empty()) {
553  // Note that the attribute names directly reference bytes in
554  // mlirAttributes, so that vector must not be changed from here
555  // on.
556  llvm::SmallVector<MlirNamedAttribute, 4> mlirNamedAttributes;
557  mlirNamedAttributes.reserve(mlirAttributes.size());
558  for (auto &it : mlirAttributes)
559  mlirNamedAttributes.push_back(
560  mlirNamedAttributeGet(it.first.c_str(), it.second));
561  mlirOperationStateAddAttributes(&state, mlirNamedAttributes.size(),
562  mlirNamedAttributes.data());
563  }
564  if (!mlirSuccessors.empty())
565  mlirOperationStateAddSuccessors(&state, mlirSuccessors.size(),
566  mlirSuccessors.data());
567  if (regions) {
568  llvm::SmallVector<MlirRegion, 4> mlirRegions;
569  mlirRegions.resize(regions);
570  for (int i = 0; i < regions; ++i)
571  mlirRegions[i] = mlirRegionCreate();
572  mlirOperationStateAddOwnedRegions(&state, mlirRegions.size(),
573  mlirRegions.data());
574  }
575 
576  // Construct the operation.
577  MlirOperation operation = mlirOperationCreate(&state);
578  return PyOperation::createDetached(getRef(), operation).releaseObject();
579 }
